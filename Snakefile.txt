# Define o arquivo de configuração global que contém caminhos e metadados
configfile: "CONFIG/config.yaml"

# Importa a biblioteca pandas para ler a tabela de metadados das amostras (samples.tsv)
import pandas as pd
samples_df = pd.read_csv("CONFIG/samples.tsv", sep="\t")
SAMPLES = samples_df["sample"].tolist()


# Define todos os arquivos finais obrigatórios que o pipeline deve entregar.
# O Snakemake avalia o pipeline de trás para frente a partir daqui.

rule all:
    input:
        "results/multiqc_report.html",
        "results/deseq2/pca_plot.png",
        "results/deseq2/functional_enrichment.png"

# Controle de Qualidade Inicial
# Roda o FastQC nos arquivos brutos para avaliar qualidade de bases e adaptadores.

rule fastqc:
    input:
        r1="data/raw/{sample}_1.fastq.gz",
        r2="data/raw/{sample}_2.fastq.gz"
    output:
        h1="results/fastqc/{sample}_1_fastqc.html",
        h2="results/fastqc/{sample}_2_fastqc.html"
    message: "Executando FastQC na amostra {wildcards.sample}..."
    shell:
        "fastqc {input.r1} {input.r2} -o results/fastqc/"


# Trimming e Filtragem de Qualidade
# Usa o fastp para remover adaptadores Illumina e bases de baixa qualidade (Phred < 20).

rule fastp:
    input:
        r1="data/raw/{sample}_1.fastq.gz",
        r2="data/raw/{sample}_2.fastq.gz"
    output:
        r1="results/fastp/{sample}_1.trimmed.fastq.gz",
        r2="results/fastp/{sample}_2.trimmed.fastq.gz",
        html="results/fastp/{sample}.html",
        json="results/fastp/{sample}.json"
    message: "Executando controle de qualidade e trimming com fastp para {wildcards.sample}..."
    shell:
        "fastp -i {input.r1} -I {input.r2} -o {output.r1} -O {output.r2} -h {output.html} -j {output.json}"


# Consolidação de Relatórios com MultiQC
# Agrupa os outputs do FastQC e do fastp em um único relatório interativo HTML.

rule multiqc:
    input:
        expand("results/fastqc/{sample}_1_fastqc.html", sample=SAMPLES),
        expand("results/fastp/{sample}.html", sample=SAMPLES)
    output:
        "results/multiqc_report.html"
    message: "Consolidando relatórios de qualidade com MultiQC..."
    shell:
        "multiqc results/fastqc/ results/fastp/ -o results/"


# Indexação do Transcriptoma de Referência
# Prepara a referência de cDNA (gencode) para que o Salmon possa fazer o alinhamento rápido.

rule salmon_index:
    input:
        fasta="reference/transcripts.fa.gz"
    output:
        directory("reference/salmon_index")
    message: "Criando índice do transcriptoma de referência com Salmon..."
    shell:
        "salmon index -t {input.fasta} -i {output} --gencode"



#  Quantificação de Expressão (Quasi-mapping)
# Mapeia as leituras limpas contra o índice criado e calcula as contagens por transcrito.

rule salmon_quant:
    input:
        r1="results/fastp/{sample}_1.trimmed.fastq.gz",
        r2="results/fastp/{sample}_2.trimmed.fastq.gz",
        index="reference/salmon_index"
    output:
        "results/salmon/{sample}/quant.sf"
    message: "Quantificando expressão gênica com Salmon para {wildcards.sample}..."
    shell:
        "salmon quant -i {input.index} -l A -1 {input.r1} -2 {input.r2} --validateMappings -o results/salmon/{sample}"


#Análise Estatística de Expressão Diferencial (R / DESeq2)
# Consome as quantificações do Salmon, importa via tximport, aplica o modelo estatístico
# do DESeq2 e plota os gráficos diagnósticos: PCA, Volcano, Heatmap e MA Plot.

rule deseq2_analysis:
    input:
        quant_files = expand("results/salmon/{sample}/quant.sf", sample=SAMPLES),
        samples_tsv = "CONFIG/samples.tsv"
    output:
        pca = "results/deseq2/pca_plot.png",
        volcano = "results/deseq2/volcano_plot.png",
        heatmap = "results/deseq2/heatmap_plot.png",
        ma = "results/deseq2/ma_plot.png",
        results_csv = "results/deseq2/MEKi_vs_ctrl.csv"
    message: "Executando análise de expressão diferencial no R com DESeq2..."
    script:
        "scripts/deseq2_analysis.R"


# Análise de Enriquecimento Funcional (R / clusterProfiler)
# Pega a lista de genes alterados gerada pelo DESeq2 e busca termos de Ontologia
# Gênica (GO) e vias metabólicas (KEGG) enriquecidas.

rule clusterprofiler_enrichment:
    input:
        results_csv = "results/deseq2/MEKi_vs_ctrl.csv"
    output:
        enrich_plot = "results/deseq2/functional_enrichment.png"
    message: "Executando análise de enriquecimento funcional com clusterProfiler..."
    script:
        "scripts/enrichment_analysis.R"
